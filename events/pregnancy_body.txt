namespace = pregnancy_body

pregnancy_body.1001 = {
	hidden = yes
	trigger = {
		has_variable = pregnancy_fat
	}
	immediate = { 
		change_current_weight = {
			value = var:pregnancy_fat
			divide = 8
		 }
	}
}

pregnancy_body.2001 = {
	hidden = yes
	trigger = {
		has_variable = post_pregnancy_breasts
	}
	immediate = { 
		if = { 
			limit = { var:post_pregnancy_breasts <= 0 }
			remove_variable = post_pregnancy_breasts
		}
		else = {
			change_variable = {
				name = post_pregnancy_breasts
				add = -2
			}
			if = {
				limit = { var:post_pregnancy_breasts <= 0 }
				remove_variable = post_pregnancy_breasts
			}
			else = {
				trigger_event = {
					id = pregnancy_body.2001
					days = 30
				}
			}
		}
	}
}
