province_event = {
	id = "cheatersbless.4"
	desc = "EVTDESCcheatersbless.4"
	picture = GFX_evt_culture_change
	
	trigger = {
		
		owner = {
			OR = {
				AND = {
					trait = x_cheater
					OR = {
						NOT = { culture = ROOT }
						NOT = { religion = ROOT }
					}
				}
				AND = {
					OR = {
						NOT = { culture = ROOT }
						NOT = { religion = ROOT }
					}
					any_liege = {
						trait = x_cheater
						culture = PREV
						religion = PREV
					}
				}
				AND = {
					top_liege = {
						trait = x_cheater
						OR = {
							NOT = { culture = ROOT }
							NOT = { religion = ROOT }
						}
					}
				}
			}
		}
		
		NOT = { # No flipping back to Norse culture before 1150
			AND = {
				owner = { culture = norse }
				ROOT = {
					OR = {
						culture = swedish
						culture = norwegian
						culture = danish
						culture = norman
					}
				}
				NOT = {	has_global_flag = norse_culture_flipping }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = "EVTOPTAcheatersbless.4"
		trigger = {
			owner = {
				OR = {
					AND = {
						trait = x_cheater
						OR = {
							NOT = { culture = ROOT }
							NOT = { religion = ROOT }
						}
					}
					AND = {
						OR = {
							NOT = { culture = ROOT }
							NOT = { religion = ROOT }
						}
						any_liege = {
							trait = x_cheater
							culture = PREV
							religion = PREV
						}
					}
				}
			}
		}
		owner = {
			ROOT = {
				if = {
					limit = {
						NOT = { culture = PREV }
					}
					culture = PREV
					}
				if = {
					limit = {
						NOT = { religion = PREV }
					}
					religion = PREV
				}
				if = {
					limit = {
						has_province_modifier = nomad_agitation
					}
					remove_province_modifier = nomad_agitation
				}
			}
		}
	}

	option = {
		name = "EVTOPTAcheatersbless.4"
		trigger = {
			owner = {
				NOT = { trait = x_cheater }
				top_liege = {
					trait = x_cheater
					OR = {
						NOT = { culture = ROOT }
						NOT = { religion = ROOT }
					}
				}
			}
		}
		owner = {
			top_liege = {
				ROOT = {
					if = {
						limit = {
							NOT = { culture = PREV }
						}
						culture = PREV
					}
					if = {
						limit = {
							NOT = { religion = PREV }
						}
						religion = PREV
					}
					if = {
						limit = {
							has_province_modifier = nomad_agitation
						}
						remove_province_modifier = nomad_agitation
					}
				}
			}
		}
	}
}
