carn_fetish.1000 = {
	hidden = yes

	trigger = {
		has_game_rule = carn_fetishes_enabled
		has_game_rule = carn_content_anal_enabled
		NOT = {
			carn_has_fetish_trigger = {
				FETISH = anal
			}
		}
	}

	weight_multiplier = {
		base = 1
		modifier = { # Male & Futa
			factor = 2
			carn_gender_can_impregnate_trigger = yes
		}
		modifier = { # Male Gay
			factor = 2
			is_male = yes
			is_attracted_to_men = yes
		}
	}

	immediate = {
		carn_add_fetish_effect = {
			FETISH = anal
		}
	}
}
