pregnancy_breasts = {
	usage = game
	selection_behavior = max
	pregnancy_breasts = {
		ignore_outfit_tags = yes
		dna_modifiers = {
			morph = {
				mode = modify
				gene = gene_bs_bust
				value = {
					value = var:pregnancy_breasts
					multiply = 0.01
					multiply = {
						add = pregnancy_days
						divide = 120 # PREGNANCY_DAYS is 280; we want 1 month at 100%
						max = 1
					}
				}
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 50
				is_alive = yes
				has_trait = pregnant
				trigger_if = {
					limit = { is_alive = yes }
					has_variable = pregnancy_breasts
				}
			}
		}
	}
}
