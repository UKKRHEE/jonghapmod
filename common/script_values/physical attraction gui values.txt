sexuality_adjusted_physical_attraction_gui_value = {
						value = 0
						add = {
							if = {
								limit = { scope:attracting_character = { is_male = yes } }
								value = male_physical_attraction_gui_value
							}
							else = {
								value = female_physical_attraction_gui_value
							}
						}
						if = {
							limit = {
								trigger_if = {
									limit = { scope:attracting_character = { is_male = yes } }
									male_physical_attraction_gui_value >= 0
								}
								trigger_if = {
									limit = { scope:attracting_character = { is_female = yes } }
									female_physical_attraction_gui_value >= 0
								}
							}
							multiply = {
								if = {
									limit = {
										sex_opposite_of = scope:attracting_character
										has_sexuality = heterosexual
									}
									value = 1
								}
								else_if = {
									limit = { 
										sex_same_as = scope:attracting_character
										has_sexuality = homosexual
									}
									value = 1
								}
								else_if = {
									limit = { has_sexuality = asexual }
									value = 0
								}
								else_if = {
									limit = { 
										has_variable = sexual_orientation
										sex_opposite_of = scope:attracting_character
									}
									value = var:sexual_orientation
									add = 100
									max = 100
									divide = 100
								}
								else_if = {
									limit = { 
										has_variable = sexual_orientation
										sex_same_as = scope:attracting_character
									}
									value = var:sexual_orientation
									multiply = -1
									add = 100
									max = 100
									divide = 100
								}
								else = {
									value = 0
								}
							}
						}
						multiply = {
							if = {
								limit = { 
									has_variable = aging_attraction_offset
									scope:attracting_character = { has_variable = visual_age }
								}
								value = 1
								subtract = {
									value = {
										value = scope:attracting_character.var:visual_age
										subtract = var:aging_attraction_offset
										min = 0
										max = 1
									}
								}
							}
							else_if = {
								limit = { 
									scope:attracting_character = { has_variable = visual_age }
								}
								value = 1
								subtract = {
									value = {
										value = scope:attracting_character.var:visual_age
										subtract = {
											value = age
											subtract = 25
											multiply = 0.02
											min = 0
											max = 1
										}
										min = 0
										max = 1
									}
								}
							}
							else_if = {
								limit = { 
								has_variable = aging_attraction_offset
								}
								value = 1
								subtract = {
									value = {
										value = scope:attracting_character.age
										subtract = 25
										multiply = 0.02
										min = 0
										max = 1
										subtract = var:aging_attraction_offset
