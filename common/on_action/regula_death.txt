### Deals with paelex.
on_death = {
	on_actions = {
		on_death_magister
	}
}
on_death_magister = {
	trigger = {
		is_ai = no
		has_trait = magister_trait_group
	}
	effect = {
		every_spouse = {
			if = {
				limit = {
					has_trait = paelex
				}
				remove_trait = paelex
				add_trait = orba
			}
		}
		every_spouse = { 
		if = {
			limit = {
				has_trait = domina
			}
			remove_trait = domina
			add_trait = orba
		}
		every_concubine = {
			if = {
				limit = {
					has_trait = paelex
				}
				remove_trait = paelex
				add_trait = orba
			}
		}
		every_concubine = { 
			if = {
				limit = {
					has_trait = domina
				}
				remove_trait = domina
				add_trait = orba
			}
		}
		root.primary_heir = {
			trigger_event = {
				id = regula_initialize_event.0010
				days = { 7 28 }
			}
		}
	}
}
