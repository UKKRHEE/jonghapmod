on_sex_variable_seeding = {
	trigger = {
		NOT = {
			is_target_in_global_variable_list = {
				name = triggered_bookmark_events
				target = flag:sex_variable_seeding
				}
			}
		}
	effect = {
		add_to_global_variable_list = {
			name = triggered_bookmark_events
			target = flag:sex_variable_seeding
		}
		every_living_character = {
				trigger_event = { on_action = on_cultural_attraction_seeding }
		}
	}
}

on_cultural_attraction_seeding = {
