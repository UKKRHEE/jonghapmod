
on_cultural_penis_size_preference_seeding = {
			trigger = {
				Not = { has_variable = penis_size_preference }
			}
			effect = {
				set_variable = {
					name = penis_size_preference
					value = {
						value = 0
						add = penis_size_preference_default
						add = {
							fixed_range = { 
								min = -10
								max = 10
							}	
						}
					}
				}
			}
		}
