on_muscle_variable = {
		effect = {
			if = {
				limit = { not = { has_variable = muscles } }
				set_variable = {
					name = muscles
					value = muscle_target
				}
			}
			else = {
				set_variable = {
					name = old_muscles
					value = var:muscles
				}
				remove_variable = muscles
				set_variable = {
					name = muscles
					value = {
						value = muscle_target
						subtract = var:old_muscles
						divide = 3
						add = var:old_muscles
					}
				}
				remove_variable = old_muscles
			}
			trigger_event = { on_action = on_add_muscle_modifier }
		}
	}
