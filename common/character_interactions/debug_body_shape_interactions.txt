
add_height_interaction = {
	category = interaction_debug_main
	use_diplomatic_range = no
	ignores_pending_interaction_block = yes
	is_shown = {
		debug_only = yes
		scope:recipient = { has_variable = height_genetic }
	}
	
	on_accept = {
		scope:recipient = { save_scope_as = modified_character }
		scope:actor = { trigger_event = body_modification.0001 }
	}
	
	auto_accept = yes
}
