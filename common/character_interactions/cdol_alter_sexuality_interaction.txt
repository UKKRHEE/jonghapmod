	interface_priority = 9
	common_interaction = yes
    
    desc = cdol_alter_sexuality_interaction_desc

		    cdol_has_succubus_powers_trigger = yes

        scope:actor = {

                VALUE = cdol_alter_sexuality_sexual_power_cost
            }
        }
    }

    on_accept = {
        scope:actor = {
            trigger_event = { id = cdol_ability.0300 }
        }
    }

    auto_accept = yes
}
