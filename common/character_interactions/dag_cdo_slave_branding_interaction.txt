dag_cdo_slave_branding_interaction = {
	interface_priority = 35

	category = interaction_category_prison

	desc = dag_cdo_slave_branding_interaction_desc

	is_shown = {
		scope:recipient = {
			OR = {
				is_at_same_location = scope:actor
				is_in_the_same_court_as_or_guest = scope:actor
				is_imprisoned_by = scope:actor
			}
			NOT = { has_character_modifier = dag_cdo_slave_branded }
		}
		scope:actor = {
			is_adult = yes
			has_relation_slave = scope:recipient
		}
	}

	is_valid_showing_failures_only = {
		scope:recipient = {
			custom_description = {
				text = "currently_being_tortured"
				NOT = { has_character_flag = is_being_tortured }
			}
		}
	}

	is_highlighted = {
		OR = {
			scope:actor = { has_trait = sadistic }
			scope:actor = { has_relation_rival = scope:recipient }
			scope:recipient = {
				any_claim = {
					holder = scope:actor
				}
			}
		}
	}

	on_accept = {
		scope:recipient = {
			add_character_flag = {
				flag = is_being_tortured
				months = 6 #~Players in MP can keep the event window open...
			}
		}

		blind_and_castrate_actor_effect = yes #~Stress & dread

		show_as_tooltip = {
			scope:recipient = {
				add_character_modifier = {
					modifier = dag_cdo_slave_branded
				}
			}
		}

		torture_blind_castrate_opinion_effect = { VERB = branded }

		scope:actor = {
			trigger_event = {
				id = dag_cdo_slave_interaction_event.1020
				days = 1
			}
		}
	}

	auto_accept = yes

	ai_targets = {
		ai_recipients = prisoners
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = -20
		ai_value_modifier = {
			ai_compassion = tiny_chance_impact_negative_ai_value #~Adds +50 for highly uncompassionate characters, -50 for highly compassionate characters
		}
		opinion_modifier = {
			opinion_target = scope:recipient
			multiplier = -0.25
		}
		modifier = {
			has_character_modifier = dag_cdo_slavers_guild_member
			add = 50
		}
		modifier = {
			has_trait = sadistic
			add = 20
		}		
	}

	ai_frequency = 36
}
