carn_increase_dick_size_one_step_effect = {
	if = {
		limit = {
			has_game_rule = carn_dt_enabled
			OR = {
				is_adult = yes
				age >=11
			}
			carn_should_have_dick_trigger = yes
		}
		if = {
			limit = { carn_has_dick_big_2_trigger = yes }
			carn_remove_dick_trait_effect = yes
