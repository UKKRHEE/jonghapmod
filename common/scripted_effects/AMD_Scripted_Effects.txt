AMD_E_SEX_UNKNOWN = {	#정체불명의 상대와 1회 한 것으로 간주한다.
	if = {	limit = {	trait = AMD_virgin	}	remove_trait = AMD_virgin	}
	if = {	limit = {	always = no	}
		if = {	limit = {	check_variable = {	which = AMD_V_FIRST_CODE	value == 0	}	}
			set_variable = {	which = AMD_V_FIRST_CODE	value = 99999	}
		}
		change_variable = {	which = AMD_V_XP_SEX	value = 1	}
		if = {	limit = {	AMD_T_IS_HUMAN = yes	}
			change_variable = {	which = AMD_V_XP_HUMAN	value = 1	}
			change_variable = {	which = AMD_V_HC_HUMAN	value = 1	}
		}
		if = {	limit = {	AMD_T_IS_HORSE = yes	}
			change_variable = {	which = AMD_V_XP_HORSE	value = 1	}
			change_variable = {	which = AMD_V_HC_HORSE	value = 1	}
		}
		if = {	limit = {	AMD_T_IS_BEAR = yes	}
			change_variable = {	which = AMD_V_XP_BEAR	value = 1	}
			change_variable = {	which = AMD_V_HC_BEAR	value = 1	}

		if = {	limit = {	AMD_T_IS_CAT = yes	}
			change_variable = {	which = AMD_V_XP_CAT	value = 1	}
			change_variable = {	which = AMD_V_HC_CAT	value = 1	}
		}
	}
	AMD_E_ACM_V_ROLL = yes
	AMD_E_SET_SEX_SKILL = yes
}
AMD_E_SET_SEX_OPINION_WITH_PREV = {	#PREV상대, 오피니언 유무를 체크하여 부여한다.
	if = {	#THIS	TO	PREV
		limit = {
			NOT = {	has_opinion_modifier = { who = PREV  modifier = AMD_opinion_had_sex }	}
		}
		opinion = {	who = PREV	modifier = AMD_opinion_had_sex	years = 9999	}
	}
	if = {	#PREV	TO	THIS
		limit = {
			NOT = {	reverse_has_opinion_modifier = { who = PREV  modifier = AMD_opinion_had_sex }	}
		}
		reverse_opinion = {	who = PREV	modifier = AMD_opinion_had_sex	years = 9999	}
	}
	if = {	#THIS	TO	PREV	(RAPED)
		limit = {
			AMD_T_STF_THIS_IS_RAPED = yes
			NOT = {	has_opinion_modifier = { who = PREV  modifier = AMD_opinion_raped_me }	}
		}
		opinion = {	who = PREV	modifier = AMD_opinion_raped_me	years = 20	}
	}
	if = {	#PREV	TO	THIS	(RAPED)
		limit = {
			AMD_T_STF_THIS_IS_RAPER = yes
			NOT = {	reverse_has_opinion_modifier = { who = PREV  modifier = AMD_opinion_raped_me }	}
		}
		reverse_opinion = {	who = PREV	modifier = AMD_opinion_raped_me	years = 20	}
	}
}

AMD_E_SCORE_CHECK = {
	event_target:AMD_ET_POV = {
		#	0인 경우, 설정된 수치가 없는 것으로 보고 AMD_E_2P_SCORING_1 을 실행.
		if = {	limit = {	check_variable = {	which = local_2P_SCORE	value == 0	}	}
			if = {	limit = {	AMD_T_STF_IS_ORGY = yes	}	AMD_E_MP_SCORING = yes	}
			else_if = {	limit = {	AMD_T_STF_IS_2P = yes	}	AMD_E_2P_SCORING_1 = yes	}
		}
		#상한선 700
		if = {	limit = {	check_variable = {	which = local_2P_SCORE	value > 700	}	}
			set_variable = {	which = local_2P_SCORE	value = 700	}
		}
		#스킬레벨 5 참가자가 없는 경우, 상한선 600

			limit = {
				check_variable = {	which = local_2P_SCORE	value > 600	}
				NOT = {	AMD_T_F_SKILL_5 = yes	}
			}
			set_variable = {	which = local_2P_SCORE	value = 600	}
		}
		#하한선 100
		if = {	limit = {	check_variable = {	which = local_2P_SCORE	value < 100	}	}
			set_variable = {	which = local_2P_SCORE	value = 100	}
		}
	}
}
