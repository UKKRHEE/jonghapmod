#~dag_cdo_start_working_as_pit_fighter_effect
#~dag_cdo_stop_working_as_pit_fighter_effect
#~dag_cdo_update_working_as_pit_fighter_effect

dag_cdo_start_working_as_pit_fighter_effect = {
	if = {
		limit = { has_trait = blademaster_3 }
		add_character_modifier = {
			modifier = dag_cdo_working_as_pit_fighter_3_modifier
		}
	}
	else_if = {
		limit = { has_trait = blademaster_2 }
		add_character_modifier = {
			modifier = dag_cdo_working_as_pit_fighter_2_modifier
		}
	}
	else_if = {
		limit = { has_trait = blademaster_1 }
		add_character_modifier = {
			modifier = dag_cdo_working_as_pit_fighter_1_modifier
		}
	}
	else = {
		add_character_modifier = {
			modifier = dag_cdo_working_as_pit_fighter_0_modifier
		}
	}	
	trigger_event = { on_action = dag_cdo_on_start_working_as_pit_fighter }
}

dag_cdo_stop_working_as_pit_fighter_effect = {
	if = {
		limit = { has_character_modifier = dag_cdo_working_as_pit_fighter_0_modifier }
		remove_character_modifier = dag_cdo_working_as_pit_fighter_0_modifier
	}
	if = {
		limit = { has_character_modifier = dag_cdo_working_as_pit_fighter_1_modifier }
		remove_character_modifier = dag_cdo_working_as_pit_fighter_1_modifier
	}
	if = {
		limit = { has_character_modifier = dag_cdo_working_as_pit_fighter_2_modifier }
		remove_character_modifier = dag_cdo_working_as_pit_fighter_2_modifier
	}
	if = {
		limit = { has_character_modifier = dag_cdo_working_as_pit_fighter_3_modifier }
		remove_character_modifier = dag_cdo_working_as_pit_fighter_3_modifier
	}
	trigger_event = { on_action = dag_cdo_on_stop_working_as_pit_fighter }
}

dag_cdo_update_working_as_pit_fighter_effect = {
	if = {
		limit = { has_character_modifier = dag_cdo_working_as_pit_fighter_0_modifier }
		remove_character_modifier = dag_cdo_working_as_pit_fighter_0_modifier
	}
	if = {
		limit = { has_character_modifier = dag_cdo_working_as_pit_fighter_1_modifier }
		remove_character_modifier = dag_cdo_working_as_pit_fighter_1_modifier
	}
	if = {
		limit = { has_character_modifier = dag_cdo_working_as_pit_fighter_2_modifier }
		remove_character_modifier = dag_cdo_working_as_pit_fighter_2_modifier
	}
	if = {
		limit = { has_character_modifier = dag_cdo_working_as_pit_fighter_3_modifier }
		remove_character_modifier = dag_cdo_working_as_pit_fighter_3_modifier
	}

	if = {
		limit = { has_trait = blademaster_3 }
		add_character_modifier = {
			modifier = dag_cdo_working_as_pit_fighter_3_modifier
		}
	}
	else_if = {
		limit = { has_trait = blademaster_2 }
		add_character_modifier = {
			modifier = dag_cdo_working_as_pit_fighter_2_modifier
		}
	}
	else_if = {
		limit = { has_trait = blademaster_1 }
		add_character_modifier = {
			modifier = dag_cdo_working_as_pit_fighter_1_modifier
		}
	}
	else = {
		add_character_modifier = {
			modifier = dag_cdo_working_as_pit_fighter_0_modifier
		}
	}
}
