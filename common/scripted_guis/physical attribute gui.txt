breast_size_gui = {
	scope = character

	is_shown = {
		is_female = yes
		is_alive = yes
		is_adult = yes
		trigger_if = {
			limit = {
				is_alive = yes
			}
			has_variable = breast_size_genetic
		}
	}
}
