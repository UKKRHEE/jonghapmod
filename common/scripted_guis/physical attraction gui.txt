ideal_male_gui = {
	scope = character
	
	is_shown = {
		is_alive = yes
		is_adult = yes
		trigger_if = {
			limit = {
				is_alive = yes
			}
			has_variable = penis_size_preference
			has_variable = male_butt_size_preference
			has_variable = male_muscle_preference
			has_variable = male_fat_preference
			has_variable = male_height_preference
		}
	}
}

male_attraction_gui = {
	scope = character
	saved_scopes = { attracting_character } 
	
	is_shown = {
		exists = yes
		exists = scope:attracting_character
		trigger_if = { 
			limit = { 
				exists = scope:attracting_character
				exists = yes
			}
			scope:attracting_character = { is_male = yes }
			is_alive = yes
			is_adult = yes
			scope:attracting_character = { is_alive = yes }
			scope:attracting_character = { is_adult = yes }
			trigger_if = {
				limit = { 
					is_alive = yes
					is_adult = yes
					scope:attracting_character = { is_alive = yes }
					scope:attracting_character = { is_adult = yes }
				}
				has_variable = breast_size_preference
				has_variable = breast_size_genetic
				scope:attracting_character = { has_variable = breast_size_preference }
				scope:attracting_character = { has_variable = breast_size_genetic }
			}
		}
	}
}
