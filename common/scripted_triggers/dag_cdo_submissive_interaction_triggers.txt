	NOT = { has_relation_submissive = $DOMINANT$ }
	OR = {

		has_relation_slave_owner = $DOMINANT$
		is_obedient = $DOMINANT$				#~No idea if this works/is used
		AND = {
			is_consort_of = $DOMINANT$
			OR = {
				AND = {
					is_male = yes
					faith = { has_doctrine = doctrine_gender_female_dominated }
				}
				AND = {
					is_female = yes
					faith = { has_doctrine = doctrine_gender_male_dominated }
				}

		}
	}
}
