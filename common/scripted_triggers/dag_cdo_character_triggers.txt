###TRIGGER LIST
#~dag_cdo_significant_to_root_trigger
#~dag_cdo_foreign_ruler_for_creation_trigger	(used by some create character effects)

dag_cdo_significant_to_root_trigger = {
	or = {
		is_close_or_extended_family_of = root
		has_any_scripted_relation = root
		is_consort_of = root
		is_courtier_of = root
	}
}

dag_cdo_foreign_ruler_for_creation_trigger = {
	in_diplomatic_range = root
	NOR = {
		faith.religion = root.faith.religion
		has_same_culture_group_as = root
	}
}
