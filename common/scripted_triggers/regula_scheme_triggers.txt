fascinare_allowed = {
	trigger_if = {
		limit = { 
			is_ai = yes
			OR = {
				has_trait = mulsa
				has_trait = paelex
				has_trait = domina
			}
		}
		scope:faith = {
			controls_holy_site_with_flag = holy_site_reg_mulsa_fascinare_flag
		}
	}
	trigger_else = {
		has_trait = magister_trait_group
	}
}

use_fascinare_secrecy_trigger = {
	scope:target = {
		NOR = {
			is_consort_of = scope:owner
			is_courtier_of = scope:owner
		}
		exists = liege
		OR = {
			is_consort_of = scope:target.liege
			is_close_family_of = scope:target.liege
			house = scope:target.house
		}
	}
}
