#근친 20살 미만 절대 거절 10살 미만으로 변경
guaranteed_under_20_incest_rejection_trigger = {
	$TARGET$ = {
		NOT = { is_consort_of = $SEDUCER$ }
		is_close_or_extended_family_of = $SEDUCER$
		age < 10 #### 20 to 10
		age < $SEDUCER$.age
		is_ai = yes #Players are allowed to choose
	}
}

#타겟이 오너의 봉신인가?
is_owner_targets_liege_trigger = {
	$TARGET$ = {
		OR = {
			target_is_vassal_or_below = $OWNER$ #당신의 봉신이 아니고
			
			#당신의 봉신의 부하가 아니면
			AND = {
				exists = liege.liege
				liege.liege = $OWNER$
			}
			AND = {
				exists = liege.liege.liege
				liege.liege.liege = $OWNER$
			}
			AND = {
				exists = liege.liege.liege.liege
				liege.liege.liege.liege = $OWNER$
			}
			AND = {
				exists = liege.liege.liege.liege.liege
				liege.liege.liege.liege.liege = $OWNER$
			}
		}
	}
}

#비밀친구 관계가 나와 내 주군의 종교에 범죄인가 체크
lolita_relation_is_deviancy_in_my_or_lieges_faith_trigger = {
	AND = {
		$CHARACTER$ = { is_adult = no }
		trait_is_shunned_or_criminal_in_my_or_lieges_faith_trigger = { TRAIT = deviant GENDER_CHARACTER = $CHARACTER$ }
	}
}
