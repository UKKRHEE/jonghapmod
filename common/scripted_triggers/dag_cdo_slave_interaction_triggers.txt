dag_cdo_can_become_legal_owner_of_slave_trigger = {
	is_adult = yes
	NOT = {
		faith = {
			has_doctrine = carn_doctrine_same_slavery_crime
			faith_hostility_level = {
				target = $SLAVE$.faith
				value < faith_hostile_level
			}
		}
	}
	NOT = {
		faith = {
			has_doctrine = carn_doctrine_other_slavery_crime
			faith_hostility_level = {
				target = $SLAVE$.faith
				value >= faith_hostile_level
			}
		}
	}
}

dag_cdo_can_become_legal_slave_of_root_trigger = {
	NOT = {
		has_character_flag = carn_cannot_be_enslaved
		has_trait_with_flag = carn_cannot_be_enslaved
	}
	save_temporary_scope_as = recipient_scope
	NOT = {
		root.faith = {
			has_doctrine = carn_doctrine_same_slavery_crime
			faith_hostility_level = {
				target = scope:recipient_scope.faith
				value < faith_hostile_level
			}
		}
	}
	NOT = {
		root.faith = {
			has_doctrine = carn_doctrine_other_slavery_crime
			faith_hostility_level = {
				target = scope:recipient_scope.faith
				value >= faith_hostile_level
			}
		}
	}
}

dag_cdo_slave_can_be_slave_trained_trigger = {
	is_capable_adult_ai = yes					#~Do not check for available because it excludes prisoners
	has_relation_slave_owner = $TRAINER$
	OR = {
		is_at_same_location = $TRAINER$
		is_in_the_same_court_as_or_guest = $TRAINER$ 
		is_imprisoned_by = $TRAINER$
	}
	NOT = { has_character_flag = dag_cdo_slave_trained }
	$TRAINER$ = { is_adult = yes }
}

dag_cdo_slave_can_be_slave_trained_accelerated_trigger = {
	is_capable_adult_ai = yes					#~Do not check for available because it excludes prisoners
	has_relation_slave_owner = $TRAINER$
	OR = {
		is_at_same_location = $TRAINER$
		is_in_the_same_court_as_or_guest = $TRAINER$ 
		is_imprisoned_by = $TRAINER$
	}
	has_character_flag = dag_cdo_slave_trained
	or = {
		NOT = {
			has_opinion_modifier = {
				target = $TRAINER$
				modifier = carn_wants_to_be_your_slave_opinion
			}		
		}
		# sub_can_set_relation_dominant_with_game_rule_trigger = { CHARACTER = $DOMINANT$ }
		#~Should also be possible if slave is able to be your submissive in game rules, but is not
	}
	$TRAINER$ = { is_adult = yes }
}
